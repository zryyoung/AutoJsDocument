                return invokeAdd(args);

        return Result.notImplemented(action);

    private Result invokeAdd(Map<String, Object> args) {

        Number b   = PluginUtils.getNotNull(args, "b");

        return new Result(Collections.singletonMap("sum", sum));
    }
}
